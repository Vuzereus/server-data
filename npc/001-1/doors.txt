001-1,55,71,0	script	#Door1	906,1,1,{
    close;

OnTouchFirst:
    setnpcdir 2;
    stopnpctimer;
    initnpctimer;
    close;

OnUnTouchAll:
    setnpcdir 4;
    initnpctimer;
    startnpctimer;
    close;

OnTouchCenter:
    warp "001-2-0", 37, 45;
    close;

OnTimer340:
    stopnpctimer;
    if (getnpcdir ("") == 2) setnpcdir 6;
    if (getnpcdir ("") == 4) setnpcdir 8;
    end;
}
