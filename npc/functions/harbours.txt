// Values:
//    2   Hook moving down.
//    4   Hook moving up.
//    6   Hook down.
//    8   Hook up.

function	script	harbourClic	{
    if (getnpcdir () == 0)
    {
        setnpcdir 2;

        initnpctimer;
        startnpctimer;
        close;
    }

    if (getnpcdir () == 6)
    {
        setnpcdir 4;

        initnpctimer;
        startnpctimer;
        close;
    }
}

function	script	harbourTimer	{
    stopnpctimer;
    if (getnpcdir () == 2) setnpcdir 6;
    if (getnpcdir () == 4) setnpcdir 0;
    end;
}
